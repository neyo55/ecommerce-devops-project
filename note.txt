# Command to build the frontend
 docker build -t frontend-ui:1.0 .

# Command to start the frontend 
 docker run -p 8088:80 frontend-ui:1.0


# Command to build the backend
docker build -t catalog-service:1.0 .

# Command to start the backend 
docker run -p 3001:3001 catalog-service:1.0



# Minikube commands
# 1. Start your local Kubernetes cluster
minikube start

# 2. Load the Node.js backend image into Minikube
minikube image load catalog-service:1.0

# 3. Load the React frontend image into Minikube
minikube image load frontend-ui:1.0

# Command to enable ingress service 
minikube addons enable ingress

# Command to enable metrics-server
minikube addons enable metrics-server

# Command to see IP address of cluster
minikube ip

# Command to build bridge between local host and internal K8s ingress controller 
minikube tunnel

# command to install Prometheus & Grafana Stack using helm, port forward and password retreival 
kubectl create namespace monitoring
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm repo update
helm install prometheus prometheus-community/kube-prometheus-stack --namespace monitoring
kubectl get pods -n monitoring
kubectl get pods -n monitoring --watch
kubectl port-forward svc/prometheus-grafana 3000:80 -n monitoring    (Keep this running in a terminal)
kubectl --namespace monitoring get secrets prometheus-grafana -o jsonpath="{.data.admin-password}" | base64 -d

# Command to push update on alertmanager after setting the alertmanager yaml settings
helm upgrade prometheus prometheus-community/kube-prometheus-stack --namespace monitoring -f alertmanager-values.yaml
